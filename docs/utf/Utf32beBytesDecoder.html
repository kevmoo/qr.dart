Utf32beBytesDecoder Class

int decode() {
  int value = utf32EncodedBytesIterator.next();
  value = (value << 8) + utf32EncodedBytesIterator.next();
  value = (value << 8) + utf32EncodedBytesIterator.next();
  value = (value << 8) + utf32EncodedBytesIterator.next();
  return value;
}
